public VoxelShape makeShape(){
	VoxelShape shape = VoxelShapes.empty();
	shape = VoxelShapes.combineAndSimplify(shape, VoxelShapes.create(0.3125, 0.0625, 0.3125, 0.6875, 0.125, 0.6875);
	shape = VoxelShapes.combineAndSimplify(shape, VoxelShapes.create(0.375, 0.0625, 0.375, 0.625, 1, 0.625);
	shape = VoxelShapes.combineAndSimplify(shape, VoxelShapes.create(0.25, 0, 0.25, 0.75, 0.0625, 0.75);

	return shape;
}
